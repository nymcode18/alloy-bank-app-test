<script>
  const outcomes = {
    'Approved': { icon: '🎉', title: 'Welcome to NorthBridge!', message: 'Your account has been approved. Your virtual card is ready and your full account details are on their way to your inbox.', css: 'approved' },
    'Manual Review': { icon: '🔍', title: 'Application Under Review', message: "Thanks for submitting your application. Our team is taking a closer look and we'll be in touch within 1–2 business days.", css: 'review' },
    'Deny': { icon: '✕', title: 'Application Unsuccessful', message: "We're sorry — we weren't able to approve your application at this time. You may reapply after 30 days.", css: 'denied' }
  };

  document.getElementById('applyForm').addEventListener('submit', async function(e) {
    e.preventDefault();
    const btn = document.getElementById('submitBtn');
    btn.textContent = 'Submitting…';
    btn.classList.add('loading');
    const formData = {
      name_first: document.getElementById('name_first').value.trim(),
      name_last: document.getElementById('name_last').value.trim(),
      email_address: document.getElementById('email_address').value.trim(),
      birth_date: document.getElementById('birth_date').value.trim(),
      document_ssn: document.getElementById('document_ssn').value.trim(),
      address_line_1: document.getElementById('address_line_1').value.trim(),
      address_line_2: document.getElementById('address_line_2').value.trim(),
      address_city: document.getElementById('address_city').value.trim(),
      address_state: document.getElementById('address_state').value.trim().toUpperCase(),
      address_postal_code: document.getElementById('address_postal_code').value.trim(),
    };
    try {
      const response = await fetch('/apply', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(formData)
      });
      const data = await response.json();
      showResult(data.outcome);
    } catch (err) {
      showResult('Error');
    }
  });

  function showResult(outcome) {
    const content = outcomes[outcome] || { icon: '⚠️', title: 'Something went wrong', message: 'Please try again or contact support.', css: 'denied' };
    document.getElementById('resultIcon').textContent = content.icon;
    document.getElementById('resultTitle').textContent = content.title;
    document.getElementById('resultMessage').textContent = content.message;
    document.getElementById('resultCard').className = `result-card ${content.css}`;
    document.getElementById('formSection').style.display = 'none';
    document.getElementById('result').style.display = 'block';
  }

  function resetForm() {
    document.getElementById('applyForm').reset();
    const btn = document.getElementById('submitBtn');
    btn.textContent = 'Submit Application →';
    btn.classList.remove('loading');
    document.getElementById('formSection').style.display = 'block';
    document.getElementById('result').style.display = 'none';
  }
</script>
